\section{The Operator}
%% ====================================================================

The sample-path bound is the fixed point of an operator
$\mathcal{T}\colon \mathbb{R}^{|\mathcal{S}|} \to \mathbb{R}^{|\mathcal{S}|}$
defined by
\begin{equation}\label{eq:operator}
  (\mathcal{T}\alpha)(s)
  \;=\; \max_{a \in A(s)}
  \Bigl[
    \underbrace{\varepsilon(s)}_{\text{cost mismatch}}
    \;+\; \underbrace{\gamma \sum_{s'} \alpha(s')\, P_M(s' \mid s, a)}_{\text{propagated error}}
    \;+\; \underbrace{H_\Delta[s+a]}_{\text{transition mismatch}}
  \Bigr],
\end{equation}
where:
\begin{itemize}
  \item $\varepsilon(s) = |h_M(s) - h_{\hat M}(s)|$ is the cost mismatch (does not depend on $\alpha$),
  \item $H_\Delta[s+a]$ is the transition mismatch on $\Vhatstar$ (does not depend on $\alpha$),
  \item $A(s) = \{\pistar(s),\, \pihatstar(s)\}$ for the $\alpha_{\max}$ variant,
        or $A(s) = \mathcal{A}$ for $\alpha_{\sup}$.
\end{itemize}

The only term that depends on $\alpha$ is the propagation
$\gamma \sum_{s'} \alpha(s') P_M(s' \mid s, a)$.

%% ====================================================================
\section{Proof That $\mathcal{T}$ Is a $\gamma$-Contraction}
%% ====================================================================

\begin{proposition}
For any $\alpha, \alpha' \in \mathbb{R}^{|\mathcal{S}|}$,
\[
  \|\mathcal{T}\alpha - \mathcal{T}\alpha'\|_\infty
  \;\le\; \gamma\, \|\alpha - \alpha'\|_\infty.
\]
\end{proposition}

\begin{proof}
Fix an arbitrary state $s \in \mathcal{S}$.  Let $a^\star$ denote the action
achieving the max in $(\mathcal{T}\alpha)(s)$, and $a'$ the action achieving
the max in $(\mathcal{T}\alpha')(s)$.

\medskip
\textbf{Step 1.} Write out $(\mathcal{T}\alpha)(s) - (\mathcal{T}\alpha')(s)$
using the definition~\eqref{eq:operator}:
\begin{align}
  (\mathcal{T}\alpha)(s) - (\mathcal{T}\alpha')(s)
  &= \max_{a \in A(s)} \Bigl[\varepsilon(s)
     + \gamma \sum_{s'} \alpha(s') P_M(s'|s,a) + H_\Delta[s+a]\Bigr]
     \notag \\
  &\quad - \max_{a \in A(s)} \Bigl[\varepsilon(s)
     + \gamma \sum_{s'} \alpha'(s') P_M(s'|s,a) + H_\Delta[s+a]\Bigr].
     \label{eq:diff}
\end{align}

\textbf{Step 2.} Apply the inequality $\max_a f(a) - \max_a g(a) \le \max_a [f(a) - g(a)]$:
\begin{align}
  (\mathcal{T}\alpha)(s) - (\mathcal{T}\alpha')(s)
  &\;\le\; \max_{a \in A(s)} \Bigl[
    \gamma \sum_{s'} \alpha(s') P_M(s'|s,a)
    - \gamma \sum_{s'} \alpha'(s') P_M(s'|s,a)
  \Bigr] \notag \\[4pt]
  &\quad\; \text{(the $\varepsilon(s)$ and $H_\Delta[s+a]$ terms cancel).}
  \label{eq:cancel}
\end{align}

\textbf{Step 3.} Factor and bound using linearity of expectation:
\begin{align}
  &= \max_{a \in A(s)}\; \gamma \sum_{s'} \bigl[\alpha(s') - \alpha'(s')\bigr]
     \, P_M(s'|s,a) \notag \\[4pt]
  &\le \max_{a \in A(s)}\; \gamma \sum_{s'} \bigl|\alpha(s') - \alpha'(s')\bigr|
       \, P_M(s'|s,a)
       \qquad \text{(triangle inequality)} \notag \\[4pt]
  &\le \max_{a \in A(s)}\; \gamma \,\|\alpha - \alpha'\|_\infty
       \underbrace{\sum_{s'} P_M(s'|s,a)}_{= 1}
       \qquad \text{(replace each $|\cdot|$ by the sup-norm)} \notag \\[4pt]
  &= \gamma\, \|\alpha - \alpha'\|_\infty.
  \label{eq:upper}
\end{align}

\textbf{Step 4.} By symmetry (swapping $\alpha$ and $\alpha'$), we also have
$(\mathcal{T}\alpha')(s) - (\mathcal{T}\alpha)(s) \le \gamma\,\|\alpha - \alpha'\|_\infty$.
Therefore
\[
  \bigl|(\mathcal{T}\alpha)(s) - (\mathcal{T}\alpha')(s)\bigr|
  \;\le\; \gamma\,\|\alpha - \alpha'\|_\infty.
\]

\textbf{Step 5.} Since this holds for every $s$, take the supremum over $s$:
\[
  \|\mathcal{T}\alpha - \mathcal{T}\alpha'\|_\infty
  \;=\; \max_s \bigl|(\mathcal{T}\alpha)(s) - (\mathcal{T}\alpha')(s)\bigr|
  \;\le\; \gamma\,\|\alpha - \alpha'\|_\infty.
  \qedhere
\]
\end{proof}

\paragraph{Key insight.} The contraction factor is exactly $\gamma$ because:
\begin{enumerate}
  \item The $\alpha$-independent terms ($\varepsilon$ and $H_\Delta$) cancel in the difference.
  \item The $\alpha$-dependent term is \emph{linear} in $\alpha$ and weighted by a probability distribution $P_M(\cdot|s,a)$.
  \item Averaging over a probability distribution cannot amplify the sup-norm: $\sum_{s'} |\alpha(s')| P(s'|s,a) \le \|\alpha\|_\infty$.
  \item The factor $\gamma < 1$ in front of the sum provides the strict contraction.
\end{enumerate}

%% ====================================================================
\section{Existence and Uniqueness of the Fixed Point}
%% ====================================================================

Since $\gamma < 1$, $\mathcal{T}$ is a strict contraction on the complete metric
space $(\mathbb{R}^{|\mathcal{S}|}, \|\cdot\|_\infty)$.  By the
\textbf{Banach fixed-point theorem}, there exists a unique
$\alpha^\star \in \mathbb{R}^{|\mathcal{S}|}$ such that
$\mathcal{T}\alpha^\star = \alpha^\star$, and the iterates
$\alpha^{k+1} = \mathcal{T}\alpha^k$ converge to $\alpha^\star$ from any
starting point.

%% ====================================================================
\section{Convergence Rate}
%% ====================================================================

Starting from $\alpha^0 = 0$, applying the contraction property $k$ times gives
\begin{equation}\label{eq:error-to-fixed}
  \|\alpha^k - \alpha^\star\|_\infty
  \;\le\; \gamma^k\, \|\alpha^0 - \alpha^\star\|_\infty.
\end{equation}

In practice we monitor the successive difference
$\|\alpha^{k+1} - \alpha^k\|_\infty$ rather than the (unknown) distance to
$\alpha^\star$.  This is justified by:
\begin{align}
  \|\alpha^{k+1} - \alpha^k\|_\infty
  &= \|\mathcal{T}\alpha^k - \mathcal{T}\alpha^{k-1}\|_\infty \notag \\
  &\le \gamma\, \|\alpha^k - \alpha^{k-1}\|_\infty \notag \\
  &\le \gamma^2\, \|\alpha^{k-1} - \alpha^{k-2}\|_\infty \notag \\
  &\;\;\vdots \notag \\
  &\le \gamma^k\, \|\alpha^1 - \alpha^0\|_\infty.
  \label{eq:successive}
\end{align}

We terminate when $\|\alpha^{k+1} - \alpha^k\|_\infty < \texttt{tol}$, i.e.\ when
\[
  \gamma^k\, \|\alpha^1 - \alpha^0\|_\infty \;\le\; \texttt{tol}.
\]

%% ====================================================================
\section{Iteration Count Estimate}
%% ====================================================================

Solving $\gamma^k \|\alpha^1 - \alpha^0\|_\infty \le \texttt{tol}$ for $k$:
\begin{align}
  \gamma^k &\le \frac{\texttt{tol}}{\|\alpha^1 - \alpha^0\|_\infty} \notag \\[6pt]
  k \log \gamma &\le \log\texttt{tol} - \log\|\alpha^1 - \alpha^0\|_\infty
  \qquad (\text{both sides negative}) \notag \\[6pt]
  k &\ge \frac{\log\texttt{tol} - \log\|\alpha^1 - \alpha^0\|_\infty}{\log \gamma}.
  \label{eq:k-exact}
\end{align}

Dropping the initial-error term (i.e.\ assuming $\|\alpha^1 - \alpha^0\|_\infty \approx 1$)
gives the back-of-the-envelope lower bound:
\[
  k \;\approx\; \frac{\log\texttt{tol}}{\log \gamma}
  \;=\; \frac{\log(10^{-10})}{\log(0.75)}
  \;=\; \frac{-23.026}{-0.2877}
  \;\approx\; 80.
\]
In practice the iteration converges in \textbf{94 steps} because
$\|\alpha^1 - \alpha^0\|_\infty > 1$, requiring additional iterations to
absorb the initial gap.  Substituting into~\eqref{eq:k-exact}:
\[
  k \;\ge\; 80 + \frac{\log\|\alpha^1 - \alpha^0\|_\infty}{|\log\gamma|}
  \;\approx\; 80 + \frac{\log\|\alpha^1 - \alpha^0\|_\infty}{0.2877}.
\]
The extra $\sim\!14$ iterations correspond to an initial gap
$\|\alpha^1 - \alpha^0\|_\infty \approx e^{14 \times 0.2877} \approx 57$.
